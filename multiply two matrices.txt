#include <stdio.h>

int main() {
    int r1, c1, r2, c2, i, j, k;
    int a[10][10], b[10][10], result[10][10];

    // Input dimensions and validate for multiplication
    printf("Enter rows/cols for Matrix 1: ");
    scanf("%d %d", &r1, &c1);
    printf("Enter rows/cols for Matrix 2: ");
    scanf("%d %d", &r2, &c2);

    if (c1 != r2) {
        printf("Error: Cols of Matrix 1 must equal Rows of Matrix 2.\n");
        return 1;
    }

    // Input elements and perform multiplication
    // ... [Input and Initialization code] ...

    // Multiply matrices using three nested loops
    for (i = 0; i < r1; i++) {
        for (j = 0; j < c2; j++) {
            result[i][j] = 0;
            for (k = 0; k < c1; k++) {
                result[i][j] += a[i][k] * b[k][j];
            }
        }
    }

}
